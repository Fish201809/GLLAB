source_group("Include Files" FILES ${include_dir})

message(STATUS ${CMAKE_COMMAND} )

find_package(Qt5Core CONFIG REQUIRED)
   


add_executable(feedback ${src_dir} ${include_dir})
set_target_properties(feedback PROPERTIES FOLDER "GL")
target_link_libraries(feedback tool)
target_link_libraries(feedback PRIVATE glfw)
target_link_libraries(feedback PRIVATE unofficial::gl3w::gl3w)
